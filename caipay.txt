1. Overview
1.1 Must-Read
Important Notes:
All API calls use HTTPS protocol
Callback interfaces use application/json format
Amounts must retain two decimal places (e.g., "100.00")
Always verify the signKey in callbacks to ensure authenticity
The system will retry failed callbacks up to 5 times
1.2 Encryption Rules
The signKey is generated using the following process:

Sort all request parameters by key in ASCII ascending order
Concatenate all non-empty values (excluding signKey itself)
Append your secret key to the concatenated string
Generate MD5 hash of the resulting string
Include this signKey in your request parameters

5. Sample Code
5.1 Java Encryption Example
Example implementation for generating the signKey:

import org.apache.commons.codec.digest.DigestUtils;
import java.util.*;

public class SignKeyGenerator {

    /**
     * Generates signKey for API requests.
     * 
     * @param params     Map of request parameters (excluding signKey).
     * @param secretKey  Merchant's secret key.
     * @return           MD5 hash of the signed string.
     */
    public static String generateSignKey(Map params, String secretKey) {
        // 1. Sort parameters by key in ASCII order
        List sortedKeys = new ArrayList<>(params.keySet());
        Collections.sort(sortedKeys);

        // 2. Concatenate non-empty values (skip signKey)
        StringBuilder concatenated = new StringBuilder();
        for (String key : sortedKeys) {
            if (!key.equals("signKey")) {
                String value = params.get(key);
                if (value != null && !value.isEmpty()) {
                    concatenated.append(value);
                }
            }
        }

        // 3. Append secret key
        String rawString = concatenated.append(secretKey).toString();

        // 4. Generate MD5 hash
        return DigestUtils.md5Hex(rawString);
    }

    public static void main(String[] args) {
        // Example parameters (include all required fields)
        Map params = new HashMap<>();
        params.put("token", "a41fd77e034d424b97e64ea42f17c544");
        params.put("tenantId", "1923067572689027074");
        params.put("amount", "2");
        params.put("customerOrderNo", "TEST1748433693");
        params.put("callbackUrl", "1111111111111");
        params.put("rechargeName", "John Doe");      // New field
        params.put("rechargeEmail", "john@example.com"); // New field
        params.put("rechargePhone", "+1234567890"); // New field

        String secretKey = "76b570d3acb79ef559983839b25614ac";

        String signKey = generateSignKey(params, secretKey);
        System.out.println("Generated signKey: " + signKey);
    }
}
5.2 PHP Encryption Example
Example implementation for generating the signKey:

<?php
/**
 * Generates signKey for API requests
 * 
 * @param array $params Request parameters (excluding signKey)
 * @param string $secretKey Merchant's secret key
 * @return string MD5 hash of the signed string
 */
function generateSignKey(array $params, string $secretKey): string {
    // 1. Sort parameters by key in ASCII order
    ksort($params);

    // 2. Concatenate non-empty values (excluding signKey)
    $concatenated = '';
    foreach ($params as $key => $value) {
        if ($key !== 'signKey' && $value !== null && $value !== '') {
            $concatenated .= $value;
        }
    }

    // 3. Append secret key
    $rawString = $concatenated . $secretKey;

    // 4. Generate MD5 hash
    return md5($rawString);
}

// Example usage
$params = [
    "token" => "a41fd77e034d424b97e64ea42f17c544",
    "tenantId" => "1923067572689027074",
    "amount" => "2",
    "customerOrderNo" => "TEST1748433693",
    "callbackUrl" => "1111111111111",
    "rechargeName" => "John Doe",
    "rechargeEmail" => "john@example.com",
    "rechargePhone" => "+1234567890"
];

$secretKey = "76b570d3acb79ef559983839b25614ac";

$signKey = generateSignKey($params, $secretKey);
echo "Generated signKey: " . $signKey;
?>
